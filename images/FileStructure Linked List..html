import java.util.Iterator;

public class FileStructure {
    
    private NLNode<FileObject> root;

    public FileStructure(String fileObjectName) throws FileObjectException {
        FileObject rootObject = new FileObject(fileObjectName);
        root = new NLNode<FileObject>(rootObject);
        if (rootObject.isDirectory()) {
            buildStructure(root);
        }
    }
    
    private void buildStructure(NLNode<FileObject> node) throws FileObjectException {
        FileObject fileObject = node.getElement();
        if (fileObject.isFile()) {
            return;
        }
        Iterator<FileObject> it = fileObject.directoryFiles();
        while (it.hasNext()) {
            FileObject childObject = it.next();
            NLNode<FileObject> childNode = new NLNode<FileObject>(childObject);
            node.addChild(childNode);
            buildStructure(childNode);
        }
    }

    public NLNode<FileObject> getRoot() {
        return root;
    }

    public Iterator<String> filesOfType(String type) {
        return new FileIterator(root, type);
    }
    
    private class FileIterator implements Iterator<String> {
        
        private Iterator<NLNode<FileObject>> nodeIterator;
        private String type;
        private String nextFileName;
        
        public FileIterator(NLNode<FileObject> node, String type) {
            nodeIterator = node.children();
            this.type = type;
            nextFileName = null;
        }

        private void findNext() {
            while (nodeIterator.hasNext() || nextFileName == null) {
                if (nextFileName != null) {
                    return;
                }
                NLNode<FileObject> node = nodeIterator.next();
                FileObject fileObject = node.getElement();
                if (fileObject.isFile() && fileObject.getExtension().equals(type)) {
                    nextFileName = fileObject.getAbsolutePath();
                } else if (fileObject.isDirectory()) {
                    nodeIterator = node.children();
                }
            }
        }
        
        @Override
        public boolean hasNext() {
            findNext();
            return nextFileName != null;
        }

        @Override
        public String next() {
            findNext();
            if (nextFileName == null) {
                throw new NoSuchElementException();
            }
            String fileName = nextFileName;
            nextFileName = null;
            return fileName;
